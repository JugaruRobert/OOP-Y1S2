Movie::Movie(string title, string genre, int year, int likes, string trailer) {
    this->setTitle(title);
    this->setGenre(genre);
    this->setYear(year);
    this->setLikes(likes);
    this->setTrailer(trailer);
}

    this->setTitle(title);

void Movie::setTitle(string title) {
    /*
    * SETTER for title
    */

    if(!title.length())
        throw Exception("Title length must not be zero");

    this->title = title;
}

void Movie::setGenre(string genre) {
    /*
    * SETTER for genre
    */

    if(!genre.length())
        throw Exception("Genre length must not be zero");

    this->genre = genre;
}

void Movie::setYear(int year) {
    /*
    * SETTER for year
    */

    this->year = year;
}

void Movie::setLikes(int likes) {
    /*
    * SETTER for likes
    */
    this->likes = likes;
}

void Movie::setTrailer(string trailer) {
    /*
